import cv2
import numpy as np

def calculate_contrast(image):
    # Convert the image to grayscale
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # Calculate the standard deviation of pixel values in the grayscale image
    std_dev = np.std(gray)

    return std_dev

def main():
    # Read an image from file
    image_path = 'C:/Users/DCL/Desktop/S3/00000243.jpg'
    image = cv2.imread(image_path)

    if image is None:
        print(f"Error: Unable to read the image from {image_path}")
        return

    # Calculate the contrast level
    contrast_level = calculate_contrast(image)

    # Print the contrast level
    print(f"Contrast Level: {contrast_level}")

    # # Display the original image
    # cv2.imshow('Original Image', image)
    # cv2.waitKey(0)
    # cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
